    \begin{itemize}

        \item Tư thế ngồi sai (cúi gập cổ, vẹo lưng, nhìn sát màn hình) về lâu dài gây các bệnh nghiêm trọng về cột sống và thị lực.
        \vspace{0.3cm}
        \item \textbf{Giải pháp cần thiết:} Một công cụ tự động nhắc nhở người dùng điều chỉnh tư thế theo thời gian thực mà không yêu cầu mua sắm thiết bị phần cứng đặc biệt.
    \end{itemize}

\section{Giới thiệu PoseAlert}
\begin{frame}{2. Giới thiệu PoseAlert}
    \textbf{PoseAlert} là ứng dụng web nhận diện tư thế qua webcam:
    \vspace{0.5cm}
    \begin{itemize}
        \item Hoạt động hoàn toàn trên trình duyệt, không gửi video lên máy chủ (đảm bảo quyền riêng tư).
        \item Nhận diện 4 trạng thái: Ngồi đúng, Cúi đầu, Vẹo lưng, Mắt quá gần màn hình.
        \item Cảnh báo tự động bằng âm thanh/popup sau 30 giây ngồi sai liên tục.
        \item Hỗ trợ bộ đếm Pomodoro (25/5 phút) và hệ thống biểu đồ Dashboard báo cáo phiên học.
    \end{itemize}
\end{frame}

\section{Kiến trúc \& Thuật toán}
\begin{frame}{4. Kiến trúc \& Thuật toán (1)}
    Ứng dụng không huấn luyện phân loại ảnh trực tiếp mà dùng thuật toán \textbf{hình học trên điểm khớp (Keypoint Heuristics)}.
    \vspace{0.5cm}
    \begin{itemize}
        \item \textbf{Phát hiện Cúi đầu:} \\ Tính khoảng cách theo trục Y từ điểm mũi đến tâm đường nối hai vai. Nếu tỷ lệ so với chiều rộng vai $< 0.5 \Rightarrow$ Cúi đầu.
        \vspace{0.3cm}
        \item \textbf{Phát hiện Mắt quá gần:} \\ Dùng khoảng cách 2 tai làm ước lượng bề ngang mặt. Nếu chiếm $> 40\%$ bề ngang camera $\Rightarrow$ Quá gần.
    \end{itemize}
\end{frame}

\begin{frame}{4. Kiến trúc \& Thuật toán (2)}
    \begin{itemize}
        \item \textbf{Phát hiện Vẹo lưng:} \\ 
        Góc hợp bởi đường nối hai vai so với phương ngang $> 12^\circ$, hoặc độ lệch tâm giữa trục vai và hông vượt quá $30\%$ $\Rightarrow$ Vẹo lưng.
        \vspace{0.5cm}
        \item \textbf{Kỹ thuật Làm mịn (Smoothing):} \\ 
        Do camera đôi khi nhiễu, ứng dụng lưu 5 khung hình gần nhất và "bầu chọn đa số" (Majority Vote) để chốt kết quả tư thế, giúp giao diện không bị chớp nháy.
    \end{itemize}
\end{frame}
