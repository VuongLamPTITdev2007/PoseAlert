\section{Giới thiệu PoseAlert}
\begin{frame}{2. Giới thiệu PoseAlert}
    \textbf{\Large PoseAlert} là ứng dụng web nhận diện tư thế thông minh qua webcam.
    \vspace{0.4cm}
    \begin{itemize}
        \item \textbf{100\% Client-side}: Xử lý hoàn toàn trên thiết bị của người dùng, không gửi video lên server (Bảo mật tối đa).
        \item \textbf{4 Trạng thái nhận diện}: Ngồi đúng, Cúi đầu, Vẹo lưng, Mắt quá gần màn hình.
        \item \textbf{Hệ thống Cảnh báo}: Phát âm thanh và Popup thông báo nếu người dùng ngồi sai liên tục trong 30 giây.
        \item \textbf{Tính năng bổ sung}: Tích hợp Đồng hồ Pomodoro (25 phút học/5 phút nghỉ) \& Dashboard Thống kê thời gian.
    \end{itemize}
\end{frame}

\section{Kiến trúc \& Thuật toán}
\begin{frame}{4. Thuật toán: Cúi đầu \& Mắt quá gần}
    Sử dụng \textbf{Thuật toán hình học trên điểm khớp (Keypoint Heuristics)} thay vì mạng Nơ-ron phân loại ảnh, giúp tốc độ nhanh và dễ cấu hình.
    \vspace{0.4cm}
    
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \begin{block}{🙇 Phát hiện Cúi đầu}
                Tính khoảng cách dọc (trục Y) từ điểm \textbf{Mũi} đến \textbf{Tâm hai vai}. \\
                Tỷ lệ so với chiều rộng của vai $< 0.5 \Rightarrow$ \textcolor{posealert}{\textbf{Cúi đầu}}.
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{👀 Phát hiện Mắt quá gần}
                Dùng khoảng cách 2 \textbf{Tai} làm tham chiếu bề ngang khuôn mặt. \\
                Nếu chiếm $> 40\%$ bề ngang Camera $\Rightarrow$ \textcolor{posealert}{\textbf{Quá gần}}.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{4. Thuật toán: Vẹo lưng \& Kỹ thuật Làm mịn}
    \begin{block}{↩️ Phát hiện Vẹo lưng}
        \begin{itemize}
            \item Góc hợp bởi đường nối hai vai so với phương ngang $> 12^\circ$.
            \item Hoặc: Độ lệch tâm ngang giữa điểm giữa vai và điểm giữa hông $> 30\%$.
        \end{itemize}
    \end{block}
    
    \begin{exampleblock}{✨ Kỹ thuật Làm mịn (Smoothing)}
        \begin{itemize}
            \item \textbf{Vấn đề:} Camera có thể bị nhiễu (noise) khung hình $\Rightarrow$ Kết quả chớp nháy.
            \item \textbf{Giải pháp:} Lưu bộ đệm 5 khung hình gần nhất. Áp dụng "Bầu chọn đa số" (Majority Vote) để chốt kết quả tư thế $\Rightarrow$ Ổn định giao diện.
        \end{itemize}
    \end{exampleblock}
\end{frame}
