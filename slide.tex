\begin{frame}{Lý do chọn đề tài}
    \begin{itemize}
        \item Thời gian ngồi trước máy tính ngày càng tăng.
        \item Ngồi sai tư thế gây đau lưng, mỏi cổ, giảm thị lực.
        \item Cần một giải pháp tự động nhắc nhở, hoạt động ngay trên trình duyệt mà không cần cài đặt phần mềm nặng nề.
    \end{itemize}
\end{frame}

\begin{frame}{Giới thiệu PoseAlert}
    \textbf{Tính năng chính:}
    \begin{itemize}
        \item Nhận diện thời gian thực bằng AI (TensorFlow.js + MoveNet).
        \item Phân tích 4 tư thế: Ngồi đúng, Cúi đầu, Vẹo lưng, Mắt quá gần.
        \item Giao diện App chuyên nghiệp (Mobile Responsive).
        \item Tích hợp Pomodoro Timer (25p/5p).
        \item Thống kê dữ liệu dạng biểu đồ (Chart.js).
        \item Hệ thống Đám mây (Firebase): Lưu lịch sử, Phòng học nhóm, Chat.
        \item Trò chơi hóa (Gamification): Chuỗi lửa 🔥 \& Nhiệm vụ 🎯.
        \item \textbf{Bảo mật:} Mọi xử lý ảnh thực hiện tại máy cá nhân (Client-side).
    \end{itemize}
\end{frame}

\begin{frame}{Công nghệ sử dụng}
    \begin{itemize}
        \item \textbf{AI \& Machine Learning}: TensorFlow.js, MoveNet Lightning.
        \item \textbf{Frontend (Giao diện)}: HTML5, CSS3, JavaScript (Modular architecture).
        \item \textbf{Backend (Đám mây)}: Firebase (Auth, Realtime Database).
        \item \textbf{Thư viện hỗ trợ}: Chart.js (vẽ biểu đồ), Web Audio API (âm thanh).
        \item \textbf{Triển khai}: GitHub Pages.
    \end{itemize}
\end{frame}

\begin{frame}{Cách thức hoạt động}
    \begin{enumerate}
        \item \textbf{Khởi tạo:} Tải mô hình MoveNet từ Google CDN (chỉ cần mạng ở lần đầu).
        \item \textbf{Phân tích:} Trích xuất tọa độ 17 điểm khớp xương.
        \item \textbf{Xử lý thuật toán:} Áp dụng logic hình học (góc nghiêng, tỷ lệ, khoảng cách) để phân loại tư thế.
        \item \textbf{Cảnh báo \& Lưu trữ:} Cảnh báo âm thanh sau 30s sai tư thế. Cuối phiên tự động lưu dữ liệu lên Firebase và cập nhật Nhiệm vụ/Chuỗi lửa.
    \end{enumerate}
\end{frame}

\begin{frame}{Kết luận \& Hướng phát triển}
    \textbf{Kết quả đạt được:}
    \begin{itemize}
        \item Hệ thống chạy mượt mà (30+ FPS) không cần GPU mạnh.
        \item Tính năng hoàn thiện, dễ dùng cho học sinh/sinh viên.
        \item Responsive tốt trên mọi thiết bị.
    \end{itemize}
    \vspace{0.5cm}
    \textbf{Hướng phát triển tương lai:}
    \begin{itemize}
        \item Nâng cấp lên mô hình 3D Pose Estimation.
        \item Làm thành tiện ích (Browser Extension).
        \item Phát triển bản Native Mobile App (React Native).
    \end{itemize}
\end{frame}
